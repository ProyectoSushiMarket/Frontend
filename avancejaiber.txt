trigger actualizar precio:
BEGIN
    SET NEW.precio = (SELECT precio FROM productos WHERE productos.id_producto = NEW.id_producto);
END

trigger establecer precio al hacer el pedido:
BEGIN
    
    DECLARE producto_precio DECIMAL(10, 3);

    SELECT precio INTO producto_precio
    FROM productos
    WHERE id_producto = NEW.id_producto;

    SET NEW.precio = producto_precio;
END
